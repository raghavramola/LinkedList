//Intersection point of two linked lists
---------------------------------------------

int intersection(Node* head1,Node* head2)
{
int count1=0,count2=0;
Node* curr1=head1,* curr2=head2;
while(curr1!=NULL)
{
count1++;
curr1=curr1->next;
}
while(curr2!=NULL)
{
count2++;
curr2=curr2->next;
}
curr1=head1;
curr2=head2;
while(curr1!=NULL&&curr2!=NULL)
{
if(curr1==curr2)
return curr1->key;
curr1=curr1->next;
curr2=curr2->next;
}
return -1;
}
